import { serve } from 'https://deno.land/std@0.168.0/http/server.ts' import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', } serve(async (req) => { if (req.method === 'OPTIONS') { return new Response('ok', { headers: corsHeaders }) } try { const { Body } = await req.json() const { stkCallback } = Body const { ResultCode, ResultDesc, CheckoutRequestID, CallbackMetadata } = stkCallback const supabaseClient = createClient( Deno.env.get('SUPABASE_URL') ?? '', Deno.env.get('SUPABASE_ANON_KEY') ?? '' ) if (ResultCode === 0) { // Payment successful const items = CallbackMetadata.Item const receiptNumber = items.find(i => i.Name === 'MpesaReceiptNumber')?.Value const amount = items.find(i => i.Name === 'Amount')?.Value const transactionDate = items.find(i => i.Name === 'TransactionDate')?.Value // Get the service request const { data: service } = await supabaseClient .from('service_requests') .select('*') .eq('mpesa_checkout_request_id', CheckoutRequestID) .single() if (service) { // Generate result based on service type let result = null if (service.service_type === 'valuation') { const currentYear = new Date().getFullYear() const age = currentYear - service.year const baseValue = 2000000 let marketValue = Math.max(300000, baseValue - (baseValue * age * 0.1)) if (service.make?.toLowerCase() === 'toyota') marketValue *= 1.1 if (service.make?.toLowerCase() === 'mercedes') marketValue *= 1.2 if (service.make?.toLowerCase() === 'bmw') marketValue *= 1.15 result = { market_value: Math.round(marketValue), age_years: age, depreciation_rate: age * 10, certificate_number: `AUTO-VAL-${service.id.substring(0, 8)}`, valuation_date: new Date().toISOString() } } else { const score = Math.floor(Math.random() * 25) + 75 result = { overall_score: score, verdict: score >= 85 ? 'Excellent' : (score >= 70 ? 'Good' : 'Fair'), certificate_number: `AUTO-INS-${service.id.substring(0, 8)}`, inspection_date: new Date().toISOString() } } // Update service request await supabaseClient .from('service_requests') .update({ payment_status: 'paid', status: 'completed', mpesa_receipt_number: receiptNumber, result: result, paid_at: new Date().toISOString(), completed_at: new Date().toISOString() }) .eq('id', service.id) // Record transaction await supabaseClient .from('mpesa_transactions') .insert({ checkout_request_id: CheckoutRequestID, result_code: ResultCode, result_desc: ResultDesc, amount: amount, mpesa_receipt_number: receiptNumber, transaction_date: new Date(parseInt(transactionDate)), phone_number: service.phone, service_id: service.id, status: 'completed' }) console.log(`✅ Payment completed: ${receiptNumber} for ${service.id}`) } } else { // Payment failed await supabaseClient .from('mpesa_transactions') .insert({ checkout_request_id: CheckoutRequestID, result_code: ResultCode, result_desc: ResultDesc, status: 'failed' }) console.log(`❌ Payment failed: ${ResultDesc}`) } return new Response(JSON.stringify({ ResultCode: 0, ResultDesc: 'Success' }), { headers: { ...corsHeaders, 'Content-Type': 'application/json' } }) } catch (error) { console.error('Callback error:', error) return new Response(JSON.stringify({ ResultCode: 1, ResultDesc: error.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } }) } })
